  background-color: #3eaf7c;
}

.parent-float-margin-fix-leftfixed-rightadaptive .right-fix {
  float: right;

}
</style>

  width: 100px;

  font-weight: bold;
  color: #fff;
  background-color: #3eaf7c;
}

.parent-float-overflow-leftfixed-rightadaptive .right {
  height: 100px;
  overflow: hidden; /* 触发bfc达到自适应 */
