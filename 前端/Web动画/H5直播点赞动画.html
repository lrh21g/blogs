    if (this.imgsList.length === 0) { return null }

    // å½“è¿è¡Œæ—¶é—´ diffTime å°äºè®¾ç½®çš„ scaleTime çš„æ—¶å€™ï¼ŒæŒ‰æ¯”ä¾‹éšç€æ—¶é—´å¢å¤§ï¼Œscale å˜å¤§ã€‚è¶…è¿‡è®¾ç½®çš„æ—¶é—´é˜ˆå€¼ï¼Œåˆ™è¿”å›æœ€ç»ˆå¤§å°ã€‚
    const basicScale = [0.6, 0.9, 1.2][getRandom(0, 2)]
    const getScale = (diffTime) => {
      // diffTime - ç™¾åˆ†æ¯”ã€‚è¡¨ç¤ºä»åŠ¨ç”»å¼€å§‹è¿è¡Œåˆ°å½“å‰æ—¶é—´è¿‡äº†å¤šé•¿æ—¶é—´ã€‚å®é™…å€¼æ˜¯ä» 0 --> 1 é€æ­¥å¢å¤§ã€‚
      // scaleTime - ç™¾åˆ†æ¯”ã€‚å›¾ç‰‡ä»å¼€å§‹æ”¾å¤§åˆ°æœ€ç»ˆå¤§å°ï¼Œæ‰€ç”¨æ—¶é•¿ã€‚
      if (diffTime < this.scaleTime) {
        return +(diffTime / this.scaleTime).toFixed(2) * basicScale
      }
      else {
        return basicScale
      }
    }

    const context = this.context
    // éšæœºè¯»å–ä¸€ä¸ªå›¾ç‰‡ï¼Œè¿›è¡Œæ¸²æŸ“
    const image = this.imgsList[getRandom(0, this.imgsList.length - 1)]
    const offset = 20 // xè½´åç§»é‡
    const basicX = this.width / 2 + getRandom(-offset, offset)
    const angle = getRandom(2, 10) // è§’åº¦ç³»æ•°
    const ratio = getRandom(10, 30) * (getRandom(0, 1) ? 1 : -1)

    // éšæœºå¹³æ»‘ X è½´åç§» - é€šè¿‡æ­£å¼¦( Math.sin )å‡½æ•°æ¥å®ç°å‡åŒ€æ›²çº¿
    const getTranslateX = (diffTime) => {
      if (diffTime < this.scaleTime) {
        // æ”¾å¤§æœŸé—´ï¼Œä¸è¿›è¡Œæ‘‡æ‘†ä½ç§»
        return basicX
      }
      else {
        return basicX + ratio * Math.sin(angle * (diffTime - this.scaleTime))
      }
    }

    // Y è½´åç§» - è¿è¡Œåç§»ä» this.height --> image.height / 2 ï¼Œå³ä»æœ€åº•éƒ¨ï¼Œè¿è¡Œåˆ°é¡¶éƒ¨ç•™ä¸‹ã€‚
    const getTranslateY = (diffTime) => {
      return (
        image.height / 2 + (this.height - image.height / 2) * (1 - diffTime)
      )
    }

    // æ·¡å‡º
    const fadeOutStage = getRandom(14, 18) / 100
    const getAlpha = (diffTime) => {
      const left = 1 - +diffTime
      if (left > fadeOutStage) {
        return 1
      }
      else {
        return 1 - +((fadeOutStage - left) / fadeOutStage).toFixed(2)
      }
    }

    return (diffTime) => {
      // diffTime : ç™¾åˆ†æ¯”ã€‚è¡¨ç¤ºä»åŠ¨ç”»å¼€å§‹è¿è¡Œåˆ°å½“å‰æ—¶é—´è¿‡äº†å¤šé•¿æ—¶é—´ã€‚å®é™…å€¼æ˜¯ä» 0 --> 1 é€æ­¥å¢å¤§ã€‚
      // diffTime ä¸º 0.4 çš„æ—¶å€™ï¼Œè¯´æ˜æ˜¯å·²ç»è¿è¡Œäº† 40% çš„æ—¶é—´
      // æ—¶é—´å·®å€¼æ»¡äº†ï¼Œå³ï¼šåŠ¨ç”»ç»“æŸäº†ï¼ˆ0 --> 1ï¼‰
      if (diffTime >= 1) { return true }

      context.save()

      const scale = getScale(diffTime)
      // const rotate = getRotate();
      const translateX = getTranslateX(diffTime)
      const translateY = getTranslateY(diffTime)
      context.translate(translateX, translateY) // åç§»
      context.scale(scale, scale) // ç¼©æ”¾
      // context.rotate(rotate * Math.PI / 180);
      context.globalAlpha = getAlpha(diffTime) // æ·¡å‡º

      // ç»˜åˆ¶
      context.drawImage(
        image,
        -image.width / 2,
        -image.height / 2,
        image.width,
        image.height,
      )
      context.restore() // æ¢å¤ç”»å¸ƒ(canvas)çŠ¶æ€ã€‚
    }
  }

  // å®æ—¶ç»˜åˆ¶æ‰«æå™¨
  // å¼€å¯å®æ—¶ç»˜åˆ¶æ‰«æå™¨ï¼Œå°†åˆ›å»ºçš„æ¸²æŸ“å¯¹è±¡æ”¾å…¥ renderList æ•°ç»„ï¼Œæ•°ç»„ä¸ä¸ºç©ºï¼Œè¯´æ˜ canvas ä¸Šè¿˜æœ‰åŠ¨ç”»ï¼Œå°±éœ€è¦ä¸åœçš„å»æ‰§è¡Œ scanï¼Œç›´åˆ° canvas ä¸Šæ²¡æœ‰åŠ¨ç”»ç»“æŸä¸ºæ­¢ã€‚
  scan() {
    this.context.clearRect(0, 0, this.width, this.height)
    this.context.fillStyle = '#f4f4f4'
    this.context.fillRect(0, 0, 200, 400)

    let index = 0
    let length = this.renderList.length

    if (length > 0) {
      requestFrame(this.scan.bind(this))
      this.scanning = true
    }
    else {
      this.scanning = false
    }

    // diffTime = (Date.now() - render.timestamp) / render.duration
    // å¦‚æœå¼€å§‹çš„æ—¶é—´æˆ³æ˜¯ 10000ï¼Œå½“å‰æ˜¯100100ï¼Œåˆ™è¯´æ˜å·²ç»è¿è¡Œäº† 100 æ¯«ç§’äº†ï¼Œå¦‚æœåŠ¨ç”»æœ¬æ¥éœ€è¦æ‰§è¡Œ 1000 æ¯«ç§’ï¼Œé‚£ä¹ˆ diffTime = 0.1ï¼Œä»£è¡¨åŠ¨ç”»å·²ç»è¿è¡Œäº† 10%ã€‚
    while (index < length) {
      const child = this.renderList[index]
      if (
        !child
        || !child.render
        || child.render.call(null, (Date.now() - child.timestamp) / child.duration)
      ) {
        // åŠ¨ç”»ç»“æŸï¼Œåˆ™åˆ é™¤è¯¥åŠ¨ç”»
        this.renderList.splice(index, 1)
        length--
      }
      else {
        // ç»§ç»­æ‰§è¡ŒåŠ¨ç”»
        index++
      }
    }
  }

  // å¼€å§‹/å¢åŠ åŠ¨ç”»
  // è°ƒç”¨ä¸€æ¬¡ start æ–¹æ³•æ¥ç”Ÿæˆæ¸²æŸ“å®ä¾‹ï¼Œæ”¾è¿›æ¸²æŸ“å®ä¾‹æ•°ç»„ã€‚
  // å¦‚æœå½“å‰æ‰«æå™¨æœªå¼€å¯ï¼Œåˆ™éœ€è¦å¯åŠ¨æ‰«æå™¨ï¼Œä½¿ç”¨äº† scanning å˜é‡ï¼Œé˜²æ­¢å¼€å¯å¤šä¸ªæ‰«æå™¨ã€‚
  start() {
    const render = this.createRender()
    const duration = getRandom(1500, 3000)
    this.renderList.push({
      render,
      duration,
      timestamp: Date.now(),
    })
    if (!this.scanning) {
      this.scanning = true
      requestFrame(this.scan.bind(this))
    }
    return this
  }
}

function requestFrame(cb) {
  return (
    window.requestAnimationFrame
    || window.webkitRequestAnimationFrame
    || function (callback) {
      window.setTimeout(callback, 1000 / 60)
    }
  )(cb)
}
